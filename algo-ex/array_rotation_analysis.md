public int[] solution(int[] A, int K) {
    int N = A.length;
    if (N == 0) return A;

    K %= N;
    reverse(A, 0, N - 1);
    reverse(A, 0, K - 1);
    reverse(A, K, N - 1);

    return A;
}

private void reverse(int[] A, int start, int end) {
    while (start < end) {
        int temp = A[start];
        A[start] = A[end];
        A[end] = temp;
        start++;
        end--;
    }
}
